public void move(Direction d) throws MovementException,NotEnoughActionsException{
			int x=getLocation().x;
			int y=getLocation().y;
			//checks if movement is valid
			if((d==Direction.LEFT && x==0)||(d==Direction.RIGHT && x==14) || (d==Direction.UP && y==14 ) || (d==Direction.DOWN && y==0)){
				throw new MovementException("you are on the edge of the map, cannot move");
			}
            
			//checks if player has enough action points
			if(actionsAvailable<=0){
				throw new NotEnoughActionsException("you do not have enough action points");
			}
			//switch cases
			switch(d){
				case UP:
				Cell up=Game.map[x][y+1];
					//check if cell has a character
					if(up instanceof CharacterCell && ((CharacterCell) up).getCharacter()!=null){
						throw new MovementException("Cell contains a character");
					}
					//check if cell contains trap and applies trap damage
					if(up instanceof TrapCell){
						setCurrentHp(getCurrentHp()-((TrapCell) up).getTrapDamage());
					}
					//check if cell contains a collectible
					if(up instanceof CollectibleCell){
						((CollectibleCell) up).getCollectible().pickUp(this);
					}
					//adds the hero to the target cell
					Game.map[x][y+1]=new CharacterCell(this);
					setLocation(new Point(x,y+1));


				case DOWN:
				Cell down=Game.map[x][y-1];
					if(down instanceof CharacterCell && ((CharacterCell) down).getCharacter()!=null){
						throw new MovementException("Cell contains a character");
					}

					if(down instanceof TrapCell){
						setCurrentHp(getCurrentHp()-((TrapCell) down).getTrapDamage());
					}
					if(down instanceof CollectibleCell){
						((CollectibleCell) down).getCollectible().pickUp(this);
					}
					Game.map[x][y-1]=new CharacterCell(this);
					setLocation(new Point(x,y-1));

				case LEFT:
				Cell left=Game.map[x-1][y];
				if(left instanceof CharacterCell && ((CharacterCell) left).getCharacter()!=null){
					throw new MovementException("Cell contains a character");
				}
				if(left instanceof TrapCell){
					setCurrentHp(getCurrentHp()-((TrapCell) left).getTrapDamage());
				}
				if(left instanceof CollectibleCell){
					((CollectibleCell) left).getCollectible().pickUp(this);
				}
				Game.map[x-1][y]=new CharacterCell(this);
				setLocation(new Point(x-1,y));
				

				case RIGHT:
				Cell right=Game.map[x+1][y];
				if(right instanceof CharacterCell && ((CharacterCell) right).getCharacter()!=null){
					throw new MovementException("Cell contains a character");
				}
				
				if(right instanceof TrapCell){
					setCurrentHp(getCurrentHp()-((TrapCell) right).getTrapDamage());
				}
				if(right instanceof CollectibleCell){
					((CollectibleCell) right).getCollectible().pickUp(this);
				}
				Game.map[x+1][y]=new CharacterCell(this);
				setLocation(new Point(x+1,y));
			}
			//removes hero from old cell
			Game.map[x][y].setCharacter(null);
			actionsAvailable--;
			

		}
